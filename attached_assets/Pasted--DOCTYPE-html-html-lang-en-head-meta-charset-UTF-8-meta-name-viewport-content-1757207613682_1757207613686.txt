    <script>
        document.addEventListener('DOMContentLoaded', () => {

            const energeticPlayfulPalette = {
                red: '#FF6B6B',
                yellow: '#FFD166',
                green: '#06D6A0',
                blue: '#118AB2',
                dark: '#073B4C'
            };

            const wrapLabel = (str, maxLen = 16) => {
                if (str.length <= maxLen) {
                    return str;
                }
                const words = str.split(' ');
                const lines = [];
                let currentLine = '';
                for (const word of words) {
                    if ((currentLine + word).length > maxLen && currentLine.length > 0) {
                        lines.push(currentLine.trim());
                        currentLine = '';
                    }
                    currentLine += `${word} `;
                }
                lines.push(currentLine.trim());
                return lines;
            };

            const CHART_TOOLTIP_CONFIG = {
                plugins: {
                    legend: {
                        labels: {
                           color: '#073B4C'
                        }
                    },
                    tooltip: {
                        callbacks: {
                            title: function(tooltipItems) {
                                const item = tooltipItems[0];
                                let label = item.chart.data.labels[item.dataIndex];
                                if (Array.isArray(label)) {
                                  return label.join(' ');
                                } else {
                                  return label;
                                }
                            }
                        }
                    }
                },
                scales: {
                     x: { ticks: { color: '#073B4C' }, grid: { display: false } },
                     y: { ticks: { color: '#073B4C' }, grid: { display: false } }
                }
            };

            const userPersonasCtx = document.getElementById('userPersonasChart');
            if(userPersonasCtx) {
                new Chart(userPersonasCtx, {
                    type: 'doughnut',
                    data: {
                        labels: ['Rural Self-Builders', 'First-Time Buyers', 'Investors', 'Relocators'],
                        datasets: [{
                            label: 'User Focus',
                            data: [40, 25, 20, 15],
                            backgroundColor: [
                                energeticPlayfulPalette.red,
                                energeticPlayfulPalette.blue,
                                energeticPlayfulPalette.green,
                                energeticPlayfulPalette.yellow
                            ],
                            borderColor: '#ffffff',
                            borderWidth: 4
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                             legend: {
                                position: 'top',
                                labels: {
                                    color: '#073B4C'
                                }
                            },
                            tooltip: CHART_TOOLTIP_CONFIG.plugins.tooltip
                        }
                    }
                });
            }
            
            const provincialDensityCtx = document.getElementById('provincialDensityChart');
            if(provincialDensityCtx) {
                new Chart(provincialDensityCtx, {
                    type: 'bar',
                    data: {
                        labels: ['Gauteng', 'KwaZulu-Natal', 'Western Cape', 'Eastern Cape', 'Limpopo', 'Mpumalanga', 'North West', 'Free State', 'Northern Cape'].map(l => wrapLabel(l)),
                        datasets: [{
                            label: 'Persons per Household',
                            data: [4.1, 3.8, 3.5, 3.9, 3.2, 3.1, 2.9, 2.8, 2.5],
                            backgroundColor: energeticPlayfulPalette.blue,
                            borderRadius: 4
                        }]
                    },
                    options: { ...CHART_TOOLTIP_CONFIG, responsive: true, maintainAspectRatio: false }
                });
            }

            const landTenureCtx = document.getElementById('landTenureChart');
            if(landTenureCtx) {
                new Chart(landTenureCtx, {
                    type: 'bar',
                    data: {
                        labels: ['Target Market Land Tenure'],
                        datasets: [
                            {
                                label: 'Informal',
                                data: [45],
                                backgroundColor: energeticPlayfulPalette.red,
                                borderRadius: 4
                            },
                            {
                                label: 'Communal',
                                data: [35],
                                backgroundColor: energeticPlayfulPalette.yellow,
                                borderRadius: 4
                            },
                             {
                                label: 'Formal',
                                data: [20],
                                backgroundColor: energeticPlayfulPalette.green,
                                borderRadius: 4
                            }
                        ]
                    },
                    options: {
                        ...CHART_TOOLTIP_CONFIG,
                        responsive: true,
                        maintainAspectRatio: false,
                        indexAxis: 'y',
                        scales: {
                            x: { stacked: true, ticks: { color: '#073B4C' }, grid: { display: false } },
                            y: { stacked: true, ticks: { color: '#073B4C' }, grid: { display: false } }
                        }
                    }
                });
            }
        });
    </script>
