        // Function to render brands
        function renderFaaBrands(searchTerm = '') {
            const container = document.getElementById('brands-container');
            let htmlContent = '';
            const lowerCaseSearchTerm = searchTerm.toLowerCase();

            for (const country in faaBrandData) {
                let countrySectionContent = ''; // Content for sectors within this country
                let countryHasVisibleContent = false;

                for (const sector in faaBrandData[country]) {
                    let sectorHtml = '';
                    let sectorHasVisibleBrands = false;
                    const brandsInSector = [];

                    // Combine brands and subnodes for searching
                    const allItemsInSector = faaBrandData[country][sector];

                    allItemsInSector.forEach(item => {
                        const lowerCaseItem = item.toLowerCase();
                        if (lowerCaseSearchTerm === '' || lowerCaseItem.includes(lowerCaseSearchTerm)) {
                            // Highlight the matching part of the item name
                            const highlightedItem = searchTerm === '' ? item : item.replace(new RegExp(searchTerm, 'gi'), match => `<span class="highlight">${match}</span>`);
                            brandsInSector.push(`<li>${highlightedItem}</li>`);
                            sectorHasVisibleBrands = true;
                        }
                    });


                    if (sectorHasVisibleBrands) {
                        sectorHtml += `
                            <div class="sector-group">
                                <h3 class="sector-title">${sector}</h3>
                                <ul class="brand-list">
                                    ${brandsInSector.join('')}
                                </ul>
                            </div>
                        `;
                        countrySectionContent += sectorHtml; // Add sector HTML to country's content
                        countryHasVisibleContent = true;
                    }
                }

                if (countryHasVisibleContent) {
                    htmlContent += `
                        <div class="country-section">
                            <h2 class="country-title">${country}</h2>
                            ${countrySectionContent}
                        </div>
                    `;
                }
            }
            container.innerHTML = htmlContent;
        }

        // Call the render function when the DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            renderFaaBrands(); // Initial render of all brands

            const searchInput = document.getElementById('brandSearchInput');
            const clearButton = document.getElementById('clearSearchButton');
            const themeToggle = document.getElementById('theme-toggle');
            const body = document.body;

            // Search functionality
            searchInput.addEventListener('input', (event) => {
                renderFaaBrands(event.target.value.trim());
            });

            clearButton.addEventListener('click', () => {
                searchInput.value = '';
