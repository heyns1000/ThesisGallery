awesome—let’s get a Gemini app running on Replit with a clean data pipeline. super short, copy-paste friendly.
1) Create the Repl
Template: Node.js
Add a file .env with:
GOOGLE_API_KEY=YOUR_GEMINI_API_KEY
PORT=3000
2) Install deps
In the Replit shell:
npm init -y
npm i express cors @google/generative-ai
3) Server (index.js)
import express from "express";
import cors from "cors";
import { GoogleGenerativeAI } from "@google/generative-ai";

const app = express();
app.use(cors());
app.use(express.json());
app.use(express.static("public"));

const genAI = new GoogleGenerativeAI(process.env.GOOGLE_API_KEY);
const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });

// chat/generate
app.post("/api/generate", async (req, res) => {
  try {
    const { prompt, context } = req.body || {};
    const result = await model.generateContent([
      { text: context || "" },
      { text: prompt || "" },
    ]);
    res.json({ ok: true, text: result.response.text() });
  } catch (e) {
    res.status(500).json({ ok: false, error: e.message });
  }
});

// simple data ingest (attach “last data” to context)
let LAST_DATA = []; // swap with DB later
app.post("/api/ingest", (req, res) => {
  const { records = [] } = req.body || {};
  LAST_DATA.push(...records);
  res.json({ ok: true, count: LAST_DATA.length });
});

app.get("/api/context", (req, res) => {
  res.json({ ok: true, context: JSON.stringify(LAST_DATA).slice(0, 5e5) });
});

app.listen(process.env.PORT || 3000, () =>
  console.log("Server up on", process.env.PORT || 3000)
);

<script>
async function getJSON(u,opts){const r=await fetch(u,opts);return r.json()}
document.getElementById('load').onclick=async()=>{
  const j=await getJSON('/api/context'); document.getElementById('ctx').textContent=j.context||'';
};
document.getElementById('run').onclick=async()=>{
  const prompt=document.getElementById('prompt').value;
  const context=document.getElementById('ctx').textContent;
  const j=await getJSON('/api/generate',{method:'POST',headers:{'Content-Type':'application/json'},body:JSON.stringify({prompt,context})});
  document.getElementById('out').textContent=j.text||j.error;
};
