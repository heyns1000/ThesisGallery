    function initPricingChart() {
        if (pricingChart) pricingChart.destroy();
        const ctx = document.getElementById('pricingChart').getContext('2d');
        const dataAnnual = reportData.pricing.chartDataAnnual;
        const dataMonthly = dataAnnual.map(p => (p / 12).toFixed(2));
        pricingChart = new Chart(ctx, {
            type: 'bar',
            data: {
                labels: reportData.pricing.chartLabels,
                datasets: [{
                    label: 'Price per Year',
                    data: dataAnnual,
                    backgroundColor: 'rgba(59, 130, 246, 0.7)',
                    borderColor: 'rgba(59, 130, 246, 1)',
                    borderWidth: 1
                }]
            },
            options: {
                maintainAspectRatio: false,
                scales: { y: { beginAtZero: true, ticks: { callback: value => '$' + value } } },
                plugins: { legend: { display: false }, tooltip: { callbacks: { label: c => `$${c.raw}` } } }
            }
        });
        document.getElementById('price-toggle').addEventListener('change', (e) => {
            const isMonthly = e.target.checked;
            pricingChart.data.datasets[0].data = isMonthly ? dataMonthly : dataAnnual;
            pricingChart.data.datasets[0].label = isMonthly ? 'Price per Month (Est.)' : 'Price per Year';
            pricingChart.options.scales.y.ticks.callback = isMonthly ? value => '$' + value : value => '$' + value;
            pricingChart.update();
        });
    }

    function navigate(event, sectionId) {
        if (event) event.preventDefault();
        window.location.hash = sectionId;
        navLinks.forEach(link => {
            link.classList.toggle('active', link.hash === `#${sectionId}`);
        });

        switch (sectionId) {
            case 'summary': renderSummary(); break;
            case 'pricing': renderPricing(); break;
            case 'marketing': renderMarketing(); break;
            case 'technical': renderTechnical(); break;
            case 'downloads': renderDownloads(); break;
            default: renderSummary();
        }
    }

    document.addEventListener('DOMContentLoaded', () => {
        const initialSection = window.location.hash.substring(1) || 'summary';
        navigate(null, initialSection);
    });
