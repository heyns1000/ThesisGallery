    <script>
        function showCheckoutStep(step) {
            document.getElementById('checkout-step-1').classList.add('hidden');
            document.getElementById('checkout-step-2').classList.add('hidden');
            document.getElementById('checkout-step-3').classList.add('hidden');
            document.getElementById(`checkout-step-${step}`).classList.remove('hidden');

            document.querySelectorAll('.checkout-step').forEach((el, index) => {
                el.classList.remove('active');
                if (index + 1 <= step) {
                    el.classList.add('active');
                }
            });
        }

        function processPayment() {
            const cardInput = document.getElementById('fakeCard');
            const paymentStatus = document.getElementById('paymentStatus');

            if (cardInput.value.trim() === '1234') {
                paymentStatus.textContent = 'Processing payment...';
                paymentStatus.classList.remove('hidden', 'text-red-500');
                paymentStatus.classList.add('text-green-500');

                setTimeout(() => {
                    paymentStatus.textContent = 'Payment successful! Getting your special receipt now.';
                    showCheckoutStep(3);
                }, 2000);
            } else {
                paymentStatus.textContent = 'Oops! Wrong secret code. Try "1234"!';
                paymentStatus.classList.remove('hidden', 'text-green-500');
                paymentStatus.classList.add('text-red-500');
            }
        }

        function resetCheckout() {
            showCheckoutStep(1);
            alert("Thanks for playing! Your new toy is being built!");
        }
    </script>
